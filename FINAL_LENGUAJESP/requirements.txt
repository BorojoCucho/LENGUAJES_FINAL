# Este proyecto solo requiere Python y la librería estándar Tkinter.
# No hay dependencias externas adicionales.

# Versión mínima de Python recomendada:
python >= 3.11

# Versión mínima de Tkinter recomendada:
tkinter >= 8.6

# Para instalar Tkinter en sistemas basados en Debian/Ubuntu:
sudo apt-get install python3-tk

# Para instalar Tkinter en sistemas basados en Red Hat/CentOS:
sudo yum install python3-tkinter
